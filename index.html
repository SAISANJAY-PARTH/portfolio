WITH stock_stats AS (
  SELECT ticker, date, close,
  AVG(close) OVER(PARTITION BY ticker
      ORDER BY date ROWS 50 PRECEDING) AS ma_50
  FROM market_ticks
)
SELECT ticker, close, ma_50,
(close - ma_50) / ma_50 AS momentum
FROM stock_stats;
